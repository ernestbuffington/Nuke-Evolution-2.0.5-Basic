Before we start with the theme edits, we recommend to do a cleanup on
your theme you are going to convert for Nuke-Evolution Basic:

1. Copy the whole content of the folder theme edits/html/themes/YOUR_THEME
   into your theme you are going to convert while keeping the folder
   structure intact (rename YOUR_THEME to the actual theme name). This will
   add necessary mod templates and images as well as the Nuke-Evolution
   specific index.html file to your theme.

2. If you run your site on a UNIX based server, you also can delete
   all files called Thumbs.db if they exist.

3. Find all files called index.htm and delete them.

4. Go into the theme edits/addons/index_html folder and
   add this file to every folder where you do not have an
   index.php file.

5. Certain themes miss some images which you may like to add
   such as the arrows used for browsing previous and next pages. To add
   those images to your theme, go into the theme edits/addons/images/arrows
   folder and depending on your theme you choose one of the available colors.
   Now upload these within the html folder and keep the folder structure
   intact.

Note: We have also provided theme edit instructions for the phpBB
      Forum ACP templates for those who like to have matching
      themes also in the phpBB Forum ACP which you will find near
      the bottom of this file.

Note: If you are supposed to ADD/REPLACE something, we suggest that you make
      sure that it matches the coding style of the specific theme you are going
      to convert. Because every themer codes differently, we cannot provide exact
      ADD and REPLACE statements for every theme. To make the themes fully compatible
      with Nuke-Evolution, try to get the edits as close as possible to the way the
      theme has been created:

      EXAMPLE: In the theme called Xtrato Veil you are supposed to add a specific line
               which will not suit the current way the theme has been created so try to
               match that style: You are supposed to do the following in regards to our
               theme edit instructions in memberlist_body.tpl:

               #
               #-----[ FIND ]------------------------------------------
               # Note: This line may vary depending on your theme. If so, search for the following part:
               # {memberrow.JOINED}
               #
                  <td class="{memberrow.ROW_CLASS}" align="center" valign="middle"><span class="gensmall">{memberrow.JOINED}</span></td>

               #
               #-----[ AFTER, ADD ]------------------------------------------
               #
                  <td class="{memberrow.ROW_CLASS}" align="center" valign="middle"><span class="gen">{memberrow.ONLINE_STATUS}</span></td>

               
               However, this theme has been coded differently so you are supposed to
               search for {memberrow.JOINED}. You should see that this line is
               different to the code snippet you are supposed to add just below that.
               For Xtrato Veil, the FIND line looks like this (see above example):

                  <td class="{memberrow.ROW_CLASS}" align="center" valign="middle"><span class="gensmall">{memberrow.JOINED}</span></td>

               Instead of adding what you see in the above AFTER, ADD statement, try to
               make it so that it matches the other lines. So add the following instead:

                  <td class="row2" align="center" valign="middle"><span class="gen">{memberrow.ONLINE_STATUS}</span></td>

               That way you will make sure that the theme will have no issues and everything
               should be aligned properly.
               
               As you may see, this is more for advanced theme converters so if you do not
               feel comfortable converting certain themes, we suggest to look for people who
               are dedicated to theme conversions.

function send_pc_dateTime() {
    var pc_dateTime = new Date()
    pc_timezoneOffset = pc_dateTime.getTimezoneOffset()*(-60);
    pc_date = pc_dateTime.getFullYear()*10000 + (pc_dateTime.getMonth()+1)*100 + pc_dateTime.getDate();
    pc_time = pc_dateTime.getHours()*3600 + pc_dateTime.getMinutes()*60 + pc_dateTime.getSeconds();

    for ( i = 0; document.links.length > i; i++ ) {
        with ( document.links[i] ) {
            if ( href.indexOf('{U_SELF}') == 0 ) {
                textLink = '' + document.links[i].firstChild.data
                if ( textLink.indexOf('http://') != 0 && textLink.indexOf('www.') != 0 && (textLink.indexOf('@') == -1 || textLink.indexOf('@') == 0 || textLink.indexOf('@') == textLink.length-1 )) {
                    if ( href.indexOf('?') == -1 ) {
                        pc_data = '?pc_tzo=' + pc_timezoneOffset + '&pc_d=' + pc_date + '&pc_t=' + pc_time;
                    } else {
                        pc_data = '&pc_tzo=' + pc_timezoneOffset + '&pc_d=' + pc_date + '&pc_t=' + pc_time;
                    }
                    if ( href.indexOf('#') == -1 ) {
                        href += pc_data;
                    } else {
                        href = href.substring(0, href.indexOf('#')) + pc_data + href.substring(href.indexOf('#'), href.length);
                    }
                }
            }
        }
    }
}
</script>
